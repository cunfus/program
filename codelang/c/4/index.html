共享内存

这玩意儿可以用来进程间通信，一直没怎么玩过，也没个实际需求用上。

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/mman.h>
#include <fcntl.h>

#define SHM_NAME "my_shm" /* /dev/shm/my_shm */
#define SHM_SIZE 256

void reader()
{
    int shm_fd = shm_open(SHM_NAME, O_RDONLY, 0666);

    char *ptr = mmap(0, SHM_SIZE, PROT_READ, MAP_SHARED, shm_fd, 0);

    printf("Read from shared memory: %s\n", ptr);

    munmap(ptr, SHM_SIZE);
    close(shm_fd);
}

void writer()
{
    int shm_fd = shm_open(SHM_NAME, O_CREAT | O_RDWR, 0666);
    ftruncate(shm_fd, SHM_SIZE);

    char *ptr = mmap(0, SHM_SIZE, PROT_WRITE, MAP_SHARED, shm_fd, 0);

    sprintf(ptr, "Hello from writer process!");

    munmap(ptr, SHM_SIZE);
    close(shm_fd);
}

int main(int argc, char **argv)
{
    if (argc != 2) {
        printf("usage: ./test r/w\n");
        exit(EXIT_FAILURE);
    }

    if ('r' == argv[1][0])
        reader();
    else
        writer();

    return 0;
}

shm 创建的空间在哪？内存里还是磁盘里？如果在内存里，创建它的进程挂了，它会怎么样？

  • ftruncate(shm_fd, SHM_SIZE); 默认内存大小是 0，这里设置好大小
  • 创建的共享内存区域直接映射到 RAM 中，所以访问速度非常快
  • 创建 shm 的进程挂了不影响，它是独立的，没人用了才被释放（引用计数，需要手动实现）
  • shm_unlink 手动清理，删除释放。

我怎么知道有没有其他人在使用共享内存？mmap 返回的是 shm 的起始地址，在这里创建引用的结构，别人要想知道，只需要查一下就行了。

typedef struct {
    int count;                          /* 引用计数 */ 
    char data[SHM_SIZE - sizeof(int)];  /* 数据区 */

那 reader 不也能操作所有数据了吗？这样不好，不如分开，一块小的 shm 用来计数 一块大的 shm 用来共享数据。还要考虑计数时是否同时在写，那还是上信号量吧。

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/mman.h>
#include <fcntl.h>
#include <semaphore.h>

#define SEM_NAME "my_sem" /* -> /dev/shm/sem.my_sem */
#define SHM_NAME "my_shm" /* -> /dev/shm/my_shm */
#define SHM_SIZE 256

void reader()
{
    int shm_fd = shm_open(SHM_NAME, O_RDWR, 0666);

    char *ptr = mmap(0, SHM_SIZE, PROT_READ, MAP_SHARED, shm_fd, 0);
    sem_t *sem = sem_open(SEM_NAME, 0);

    sem_wait(sem);
    sleep(30);
    printf("Read from shared memory: %s\n", ptr);
    sem_post(sem);

    munmap(ptr, SHM_SIZE);
    close(shm_fd);
    sem_close(sem);
}

void writer()
{
    int shm_fd = shm_open(SHM_NAME, O_CREAT | O_RDWR, 0666);
    ftruncate(shm_fd, SHM_SIZE);

    char *ptr = mmap(0, SHM_SIZE, PROT_WRITE, MAP_SHARED, shm_fd, 0);

    sem_t *sem = sem_open(SEM_NAME, O_CREAT, 0666, 1);  // creat sem = 1

    sem_wait(sem);  // sem - 1
    sprintf(ptr, "Hello from writer process!");
    sem_post(sem);  // sem + 1

    munmap(ptr, SHM_SIZE);
    close(shm_fd);
    sem_close(sem);
}

int lookup()
{
    sem_t *sem = sem_open(SEM_NAME, 0);

    int v;
    sem_getvalue(sem, &v);

    printf("sem value : %d", v);
    sem_close(sem);

    return v;
}

void cleanup()
{
    if (1 == lookup()) {
        shm_unlink(SHM_NAME);
        sem_unlink(SEM_NAME);
    } else {
        printf("someone here is using.\n");
    }
}

int main(int argc, char **argv)
{
    if (argc != 2) {
        printf("usage: ./test r/w\n");
        exit(EXIT_FAILURE);
    }

    switch (argv[1][0]) {
    case 'r':
        reader();
        break;
    case 'w':
        writer();
        break;
    case 'l':
        lookup();
        break;
    case 'c':
        cleanup();
        break;
    default:
        printf("Invalid option: %c\n", argv[1][0]);
        exit(EXIT_FAILURE);

    return 0;
}
