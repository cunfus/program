IO 复用

了解 IO 复用，先来看下简单的服务器程序。

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>

#define PORT 8080
#define BUFFER_SIZE 1024

int main() {
    int server_fd, new_socket;
    struct sockaddr_in address;
    int addrlen = sizeof(address);
    char buffer[BUFFER_SIZE] = {0};

    if ((server_fd = socket(AF_INET, SOCK_STREAM, 0)) == 0) {
        perror("Socket failed");
        exit(EXIT_FAILURE);
    }

    address.sin_family = AF_INET;
    address.sin_addr.s_addr = INADDR_ANY;
    address.sin_port = htons(PORT);

    if (bind(server_fd, (struct sockaddr *)&address, sizeof(address)) < 0) {
        perror("Bind failed");
        close(server_fd);
        exit(EXIT_FAILURE);
    }

    if (listen(server_fd, 3) < 0) {
        perror("Listen failed");
        close(server_fd);
        exit(EXIT_FAILURE);
    }

    printf("Listening on port %d...\n", PORT);

    if ((new_socket = accept(server_fd, (struct sockaddr *)&address, (socklen_t*)&addrlen)) < 0) {
        perror("Accept failed");
        close(server_fd);
        exit(EXIT_FAILURE);
    }

    printf("Connection accepted.\n");

    int valread = read(new_socket, buffer, BUFFER_SIZE);
    printf("Message from client: %s\n", buffer);

    const char *reply = "Hello from server!";
    send(new_socket, reply, strlen(reply), 0);
    printf("Reply sent to client.\n");

    close(new_socket);
    close(server_fd);
    return 0;
}

客户端 nc localhost 8080。但这个例子只能处理一个客户端，随后关闭连接了，可以上多线程，每来一个连接，创建一个线程，去处理。

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <pthread.h>

#define PORT 8080
#define BUFFER_SIZE 1024

void *handle_client(void *socket_desc) {
    int sock = *(int *)socket_desc;
    char buffer[BUFFER_SIZE] = {0};

    int valread = read(sock, buffer, BUFFER_SIZE);
    printf("Message from client: %s\n", buffer);

    const char *reply = "Hello from server!";
    send(sock, reply, strlen(reply), 0);
    printf("Reply sent to client.\n");

    close(sock);
    return NULL;
}

int main() {
    int server_fd, new_socket;
    struct sockaddr_in address;
    int addrlen = sizeof(address);
    pthread_t thread_id;

    if ((server_fd = socket(AF_INET, SOCK_STREAM, 0)) == 0) {
        perror("Socket failed");
        exit(EXIT_FAILURE);
    }

    address.sin_family = AF_INET;
    address.sin_addr.s_addr = INADDR_ANY;
    address.sin_port = htons(PORT);

    if (bind(server_fd, (struct sockaddr *)&address, sizeof(address)) < 0) {
        perror("Bind failed");
        close(server_fd);
        exit(EXIT_FAILURE);
    }

    if (listen(server_fd, 3) < 0) {
        perror("Listen failed");
        close(server_fd);
        exit(EXIT_FAILURE);
    }

    printf("Listening on port %d...\n", PORT);

    while (1) {
        new_socket = accept(server_fd, (struct sockaddr *)&address, (socklen_t*)&addrlen);
        if (new_socket < 0) {
            perror("Accept failed");
            continue;
        }

        printf("Connection accepted.\n");

        if (pthread_create(&thread_id, NULL, handle_client, (void *)&new_socket) != 0) {
            perror("Could not create thread");
        }

        pthread_detach(thread_id);
    }

    close(server_fd);
    return 0;
}

可是线程虽然好，也不能无限制使用啊，还得改，如果有多台机器还能分担下，单台机器怎么弄呢？一个线程处理完一个连接就关闭，太奢侈了，得让它多处理。

select 轮询

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <sys/select.h>
#include <errno.h>

#define PORT 8080
#define MAX_CLIENTS 10
#define BUFFER_SIZE 1024

int main() {
    int server_fd, new_socket, client_sockets[MAX_CLIENTS];
    struct sockaddr_in address;
    int addrlen = sizeof(address);
    char buffer[BUFFER_SIZE];
    fd_set readfds;

    for (int i = 0; i < MAX_CLIENTS; i++) {
        client_sockets[i] = 0;
    }

    if ((server_fd = socket(AF_INET, SOCK_STREAM, 0)) == 0) {
        perror("Socket failed");
        exit(EXIT_FAILURE);
    }

    address.sin_family = AF_INET;
    address.sin_addr.s_addr = INADDR_ANY;
    address.sin_port = htons(PORT);

    if (bind(server_fd, (struct sockaddr *)&address, sizeof(address)) < 0) {
        perror("Bind failed");
        close(server_fd);
        exit(EXIT_FAILURE);
    }

    if (listen(server_fd, 3) < 0) {
        perror("Listen failed");
        close(server_fd);
        exit(EXIT_FAILURE);
    }

    printf("Listening on port %d...\n", PORT);

    while (1) {
        // 清空文件描述符集合
        FD_ZERO(&readfds);
        FD_SET(server_fd, &readfds);
        int max_sd = server_fd;

        // 将客户端套接字加入集合
        for (int i = 0; i < MAX_CLIENTS; i++) {
            int sd = client_sockets[i];
            if (sd > 0) {
                FD_SET(sd, &readfds);
            }
            if (sd > max_sd) {
                max_sd = sd;
            }
        }

        // 等待活动套接字
        int activity = select(max_sd + 1, &readfds, NULL, NULL, NULL);

        if ((activity < 0) && (errno != EINTR)) {
            printf("Select error");
        }

        // 处理新连接
        if (FD_ISSET(server_fd, &readfds)) {
            new_socket = accept(server_fd, (struct sockaddr *)&address, (socklen_t*)&addrlen);
            printf("New connection accepted.\n");

            // 将新套接字加入数组
            for (int i = 0; i < MAX_CLIENTS; i++) {
                if (client_sockets[i] == 0) {
                    client_sockets[i] = new_socket;
                    break;
                }
            }
        }

        // 处理客户端请求
        for (int i = 0; i < MAX_CLIENTS; i++) {
            int sd = client_sockets[i];
            if (FD_ISSET(sd, &readfds)) {
                int valread = read(sd, buffer, BUFFER_SIZE);
                if (valread == 0) {
                    // 连接关闭
                    getpeername(sd, (struct sockaddr*)&address, (socklen_t*)&addrlen);
                    printf("Host disconnected, ip: %s, port: %d\n",
                           inet_ntoa(address.sin_addr), ntohs(address.sin_port));
                    close(sd);
                    client_sockets[i] = 0;
                } else {
                    buffer[valread] = '\0';
                    printf("Message from client: %s\n", buffer);
                    send(sd, buffer, valread, 0); // Echo back
                }
            }
        }
    }

    close(server_fd);
    return 0;
}

上面的每次循环，都要清空集合，然后重新加入连接描述符，太蠢了，能不能灵活些？

poll 轮询

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <poll.h>

#define PORT 8080
#define MAX_CLIENTS 10
#define BUFFER_SIZE 1024

int main() {
    int server_fd, new_socket;
    struct sockaddr_in address;
    int addrlen = sizeof(address);
    char buffer[BUFFER_SIZE];

    if ((server_fd = socket(AF_INET, SOCK_STREAM, 0)) == 0) {
        perror("Socket failed");
        exit(EXIT_FAILURE);
    }

    address.sin_family = AF_INET;
    address.sin_addr.s_addr = INADDR_ANY;
    address.sin_port = htons(PORT);

    if (bind(server_fd, (struct sockaddr *)&address, sizeof(address)) < 0) {
        perror("Bind failed");
        close(server_fd);
        exit(EXIT_FAILURE);
    }

    if (listen(server_fd, 3) < 0) {
        perror("Listen failed");
        close(server_fd);
        exit(EXIT_FAILURE);
    }

    printf("Listening on port %d...\n", PORT);

    // 设置 poll 结构
    struct pollfd fds[MAX_CLIENTS + 1]; // +1 for server_fd
    for (int i = 0; i < MAX_CLIENTS + 1; i++) {
        fds[i].fd = -1;
        fds[i].events = POLLIN;
    }
    fds[0].fd = server_fd; // 服务器套接字

    while (1) {
        // 调用 poll
        int poll_count = poll(fds, MAX_CLIENTS + 1, -1);
        if (poll_count < 0) {
            perror("Poll error");
            break;
        }

        // 处理新连接
        if (fds[0].revents & POLLIN) {
            new_socket = accept(server_fd, (struct sockaddr *)&address, (socklen_t*)&addrlen);
            printf("New connection accepted.\n");

            // 将新套接字添加到 pollfd 数组
            for (int i = 1; i < MAX_CLIENTS + 1; i++) {
                if (fds[i].fd == -1) {
                    fds[i].fd = new_socket;
                    break;
                }
            }
        }

        // 处理客户端请求
        for (int i = 1; i < MAX_CLIENTS + 1; i++) {
            if (fds[i].fd > 0 && (fds[i].revents & POLLIN)) {
                int valread = read(fds[i].fd, buffer, BUFFER_SIZE);
                if (valread == 0) {
                    // 连接关闭
                    getpeername(fds[i].fd, (struct sockaddr*)&address, (socklen_t*)&addrlen);
                    printf("Host disconnected, ip: %s, port: %d\n",
                           inet_ntoa(address.sin_addr), ntohs(address.sin_port));
                    close(fds[i].fd);
                    fds[i].fd = -1; // Mark as unused
                } else {
                    buffer[valread] = '\0';
                    printf("Message from client: %s\n", buffer);
                    send(fds[i].fd, buffer, valread, 0); // Echo back
                }
            }
        }
    }

    close(server_fd);
    return 0;
}

轮询效率不行啊，有没有别的方法？

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <sys/epoll.h>

#define PORT 8080
#define MAX_EVENTS 10
#define BUFFER_SIZE 1024

int main() {
    int server_fd, new_socket;
    struct sockaddr_in address;
    int addrlen = sizeof(address);
    char buffer[BUFFER_SIZE];

    if ((server_fd = socket(AF_INET, SOCK_STREAM, 0)) == 0) {
        perror("Socket failed");
        exit(EXIT_FAILURE);
    }

    address.sin_family = AF_INET;
    address.sin_addr.s_addr = INADDR_ANY;
    address.sin_port = htons(PORT);

    if (bind(server_fd, (struct sockaddr *)&address, sizeof(address)) < 0) {
        perror("Bind failed");
        close(server_fd);
        exit(EXIT_FAILURE);
    }

    if (listen(server_fd, 3) < 0) {
        perror("Listen failed");
        close(server_fd);
        exit(EXIT_FAILURE);
    }

    printf("Listening on port %d...\n", PORT);

    // 创建 epoll 实例
    int epoll_fd = epoll_create1(0);
    struct epoll_event ev, events[MAX_EVENTS];

    // 将服务器套接字添加到 epoll 监控中
    ev.events = EPOLLIN;
    ev.data.fd = server_fd;
    epoll_ctl(epoll_fd, EPOLL_CTL_ADD, server_fd, &ev);

    while (1) {
        // 等待事件发生
        int num_events = epoll_wait(epoll_fd, events, MAX_EVENTS, -1);
        for (int i = 0; i < num_events; i++) {
            // 处理新连接
            if (events[i].data.fd == server_fd) {
                new_socket = accept(server_fd, (struct sockaddr *)&address, (socklen_t*)&addrlen);
                printf("New connection accepted.\n");

                // 将新套接字添加到 epoll 监控中
                ev.events = EPOLLIN;
                ev.data.fd = new_socket;
                epoll_ctl(epoll_fd, EPOLL_CTL_ADD, new_socket, &ev);
            } else {
                // 处理客户端消息
                int valread = read(events[i].data.fd, buffer, BUFFER_SIZE);
                if (valread == 0) {
                    // 连接关闭
                    getpeername(events[i].data.fd, (struct sockaddr*)&address, (socklen_t*)&addrlen);
                    printf("Host disconnected, ip: %s, port: %d\n",
                           inet_ntoa(address.sin_addr), ntohs(address.sin_port));
                    close(events[i].data.fd);
                } else {
                    buffer[valread] = '\0';
                    printf("Message from client: %s\n", buffer);
                    send(events[i].data.fd, buffer, valread, 0); // Echo back
                }
            }
        }
    }

    // 关闭服务器套接字
    close(server_fd);
    return 0;
}
  • epoll_create1 是 epoll_create 的改进版本
  • close(events[i].data.fd) 关闭一个文件描述符后，该文件描述符对应的资源将被释放，并且这个位置可以被重新使用

LT（Level Triggered）:在 LT 模式下，只要文件描述符的状态满足条件（例如，有数据可读），epoll_wait 将持续返回该事件。只要有数据可用，可以多次读取同一文件描述符，直到没有数据为止。

  • 优点：
    • 简单易懂，容易实现
    • 可以在任何时候检查文件描述符的状态，无需担心漏掉事件
  • 缺点：
    • 如果不及时读取数据，可能导致频繁的事件通知，影响性能

ET（Edge Triggered）：在 ET 模式下，只有当文件描述符的状态发生变化时（如变为可读或可写），epoll_wait 才会返回该事件。若文件描述符的状态没有变化，即使有数据可读，也不会再次触发。

  • 优点：
    • 更高效，减少了系统调用次数，尤其是在高并发场景下
    • 避免了大量的重复事件通知
  • 缺点：
    • 实现复杂，需要确保在事件发生后尽快读取所有可用数据
    • 如果未能读取完所有数据，可能会错过后续的可读事件
